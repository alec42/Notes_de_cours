\section{Distribution uniforme de décès}
\begin{align*}
\ddot{a}^{(m)} &= \alpha(m) \times \ddot{a}_x + \beta(m) \\
\alpha(m) &= \frac{i \times d}{i^{(m)} \times d^{(m)}} \\
\beta(m) &= \frac{i - i^{(m)}}{i^{(m)} \times d^{(m)}}
\end{align*}
